93 void Command::sendCommand(boost::asio::serial_port * portp)
94 {
95  // first send a DLE
96  unsigned char c = Reader::DLE;
97  bs::write(*portp, bs::buffer(&c,1));
98 
99  // iterate through the buffer... looking for byte stuffing
100  unsigned int idx = 0;
101  while(getByte(c, idx)) {
102  if(c == Reader::DLE) {
103  // if the character is a DLE, stuff an extra DLE before it....
104  bs::write(*portp, bs::buffer(&c,1));
105  }
106  bs::write(*portp, bs::buffer(&c,1));
107  idx++;
108  }
109 
110  // last send a DLE ETX
111  c = Reader::DLE;
112  bs::write(*portp, bs::buffer(&c,1));
113  c = Reader::ETX;
114  bs::write(*portp, bs::buffer(&c,1));
115 }

117 Report * Reader::readStream()
118 {
119  if(!reader_open) {
120  return NULL;
121  }
122 
123  int bufidx = 0;
124  bool got_packet = false;
125  while(!got_packet) {
126  unsigned char c;
127  size_t len = bs::read(*tbolt_port,
128  bs::buffer(&c, 1));
129  if(len == 1) {
130  switch (parse_state) {
131  case INIT_WAIT_FOR_NORMCHAR:
132  if ((c != DLE) && (c != ETX)) {
133  parse_state = INIT_WAIT_FOR_DLE;
134  }
135  break;
136  case INIT_WAIT_FOR_DLE:
137  if (c == DLE) {
138  parse_state = INIT_WAIT_FOR_ETX;
139  }
140  break;
141  case INIT_WAIT_FOR_ETX:
142  if (c != ETX) {
143  parse_state = INIT_WAIT_FOR_DLE;
144  }
145  else {
146  parse_state = WAIT_DLE;
147  }
148  break;
149 
150  case WAIT_DLE:
151  if(c == DLE) {
152  parse_state = IN_PACKET_SAW_DLE;
153  }
154 
155  break;
156  case IN_PACKET:
157  if(c == DLE) {
158  parse_state = IN_PACKET_SAW_DLE;
159  }
160  else {
161  inbuf[bufidx++] = c;
162  }
163  break;
164  case IN_PACKET_SAW_DLE:
165  if(c == ETX) {
166  got_packet = true;
167  parse_state = WAIT_DLE;
168  }
169  else {
170  inbuf[bufidx++] = c;
171  parse_state = IN_PACKET;
172  }
173  break;
174  }
175  }
176  }
177 
178  // Now go through each report and see if it parses
179  for(std::list<Report*>::iterator ip = replist.begin();
180  ip != replist.end();
181  ++ip) {
182  // if the report parses, return a pointer to it.
183  if((*ip)->parseBuf(inbuf, bufidx)) return (*ip);
184  }
185 
186  buflen = bufidx;
187 
188  return NULL;
189 }
