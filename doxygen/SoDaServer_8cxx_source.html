153 int doWork(SoDa::Params & params)
154 {
156  SoDa::Debug d(params.getDebugLevel(), "SoDaServer");
157  d.setDefaultLevel(params.getDebugLevel());
158 
159  // These are the mailboxes that connect
160  // the various widgets
161  // the rx and tx streams are vectors of complex floats.
162  // we don't declare the extent here, as it will be set
163  // by a negotiation.
164  SoDa::DatMBox rx_stream, tx_stream, if_stream, cw_env_stream;
165  SoDa::CmdMBox cmd_stream(false);
166  // create a separate gps stream to avoid "leaks" and latency problems...
167  SoDa::CmdMBox gps_stream(false);
168  SoDa::CmdMBox cwtxt_stream(false);
169 
170  SoDa::SoDaThread * ctrl;
171  SoDa::SoDaThread * rx;
172  SoDa::SoDaThread * tx;
173 
174  if(params.isRadioType("USRP")) {
175 #if HAVE_UHD
176  ctrl = new SoDa::USRPCtrl(&params, &cmd_stream);
179  rx = new SoDa::USRPRX(&params, ((SoDa::USRPCtrl *)ctrl)->getUSRP(), &rx_stream, &if_stream, &cmd_stream);
180  tx = new SoDa::USRPTX(&params, ((SoDa::USRPCtrl *)ctrl)->getUSRP(), &tx_stream, &cw_env_stream, &cmd_stream);
181 #else
182  std::cerr << "lib UHD support not included in this build.\n^C to exit.\n";
183  exit(-1);
184 #endif
185  }
186  else {
187  std::cerr << boost::format("Radio type [%s] is not yet supported\nHit ^C to exit.\n") % params.getRadioType();
188  exit(-1);
189  }
190 
191 
195  SoDa::AudioALSA audio_ifc(params.getAudioSampleRate(),
196  SoDa::AudioIfc::FLOAT,
197  params.getAFBufferSize(),
198  params.getAudioPortName());
202  SoDa::BaseBandRX bbrx(&params, &rx_stream, &cmd_stream, &audio_ifc);
203  SoDa::BaseBandTX bbtx(&params, &tx_stream, &cmd_stream, &audio_ifc);
204 
206  SoDa::CWTX cwtx(&params, &cwtxt_stream, &cw_env_stream, &cmd_stream);
207 
209  SoDa::UI ui(&params, &cwtxt_stream, &rx_stream, &if_stream, &cmd_stream, &gps_stream);
210 
211 #if HAVE_GPSLIB
212  SoDa::GPSmon gps(&params, &gps_stream);
213 #endif
214 
215  d.debugMsg("Created units.");
216 
217  // Now start each of the activities -- they may or may not
218  // implement the "start" method -- not all objects need to be threads.
219 
220  d.debugMsg("Starting UI");
221  // start the UI -- configure stuff and all that.
222  ui.start();
223  d.debugMsg("Starting radio units");
224  // start command consumers first.
225  ctrl->start();
226  rx->start();
227  tx->start();
228  bbrx.start();
229  bbtx.start();
230  cwtx.start();
231 
232  // now the gps...
233 #if HAVE_GPSLIB
234  d.debugMsg("Starting gps");
235  gps.start();
236 #endif
237 
238  // wait for the user interface to tell us that it is time to quit.
239  ui.join();
240  ctrl->join();
241  rx->join();
242  tx->join();
243  bbrx.join();
244  bbtx.join();
245  cwtx.join();
246 #if HAVE_GPSLIB
247  gps.join();
248 #endif
249  d.debugMsg("Exit");
250 
251  // when we get here, we are done... (UI should not return until it gets an "exit/quit" command.)
252 
253  return 0;
254 }
