40  template<typename T> T convertFromBuf(unsigned char * buf, int pos) {
41  T ret;
42  int i;
43  int Nbytes = sizeof(T);
44  unsigned char * revp = ((unsigned char*) &ret);
45  for(i = Nbytes - 1; i >= 0; i--) {
46  revp[i] = buf[pos++];
47  }
48  return ret;
49  }
