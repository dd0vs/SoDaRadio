92 void add_sig(complex<float> * in, float freq, int len, float acc)
93 {
94  int i;
95  float phase_step = 2.0 * M_PI * freq / ((float) SAMPLE_RATE);
96  float ang = 0.0;
97  for(i = 0; i < len; i++) {
98  in[i] = acc * in[i] + complex<float>(cos(ang),sin(ang));
99  ang += phase_step;
100  if(ang > M_PI) ang = ang - (2.0 * M_PI);
101  }
102 }
