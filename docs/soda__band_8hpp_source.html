160  void saveBands(QSettings * set_p) {

162  BandMapIterator bmi(*this);
163  int i = 0;
164  while(bmi.hasNext()) {
165  bmi.next();
166  if(bmi.value().name() == "") continue;
167  set_p->setArrayIndex(i);

169  i++;
