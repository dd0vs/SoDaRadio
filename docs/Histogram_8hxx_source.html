58  for(unsigned int i = 0; i < num_buckets; i++) table[i] = 0;

61  void updateTable(double v) {
62  int idx = lround((v - min) * recip_bucket_interval);
63  if(v > max) table[num_buckets - 1]++;
64  else if(v < min) table[0]++;
65  else table[idx]++;
66  }

77  }

79  of.close();
80  }
