133 void GUISoDa::LogTable::readLogReport(const QString & fname)
134 {
135  // load the table from the input stream.
136  qDebug() << QString("Reading log file [%1]").arg(fname);
137  QFile infile(fname);
138 
139  if(!infile.open(QFile::ReadOnly)) {
140  QMessageBox::information(this, tr("Unable to open file for reading"),
141  infile.errorString());
142  return;
143  }
144 
145  QString lbuf;
146  QTextStream instr(&infile);
147  int row_start = next_used_row;
148  int max_row = -1;
149  while(!(lbuf = instr.readLine()).isNull()) {
150  int co0 = lbuf.indexOf(':');
151  int co1 = lbuf.indexOf(':',co0+1);
152 
153  // strip out row, col.
154  int row = lbuf.mid(0,co0).toInt() + row_start;
155  int col = lbuf.mid(co0+1,co1 - (co0+1)).toInt();
156  QString val = lbuf.mid(co1+1);
157  if((row + 10) > rowCount()) setRowCount(rowCount() + 20);
158  setItem(row, col, new QTableWidgetItem(val));
159  if(row > max_row) max_row = row;
160  }
161  next_used_row = max_row + 1;
162  infile.close();
163 }
